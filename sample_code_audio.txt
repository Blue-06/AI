import streamlit as st
import random
import httpx
import torch

# Fix Streamlit file-watcher crash with PyTorch
torch.classes.__path__ = []

from dotenv import load_dotenv
import os

load_dotenv()
from langchain_openai import ChatOpenAI

# -----------------------------------
# PAGE CONFIG
# -----------------------------------
st.set_page_config(
    page_title="Dynamic Voice Profile Adaptation",
    layout="wide"
)
st.title("🎤 Dynamic Voice Profile Adaptation")
st.write("AI-powered voice authentication that adapts to temporary health changes.")

# -----------------------------------
# TCS GenAI API
# -----------------------------------
api_key = os.getenv("OPENAI_API_KEY")
if not api_key:
    st.sidebar.error("❌ OPENAI_API_KEY not found in .env")
    st.stop()

client = httpx.Client(verify=False)
llm = ChatOpenAI(
    base_url="https://genailab.tcs.in",
    model="azure_ai/genailab-maas-DeepSeek-R1",
    api_key=api_key,
    http_client=client
)

# -----------------------------------
# SAMPLE API TEST
# -----------------------------------
try:
    response = llm.invoke("Hi")
    st.sidebar.success("✅ TCS GenAI Connected")
    st.sidebar.write("LLM Response:")
    st.sidebar.write(response.content)
except Exception as e:
    st.sidebar.error("❌ Failed to connect to TCS GenAI")
    st.sidebar.code(str(e))

# -----------------------------------
# SIDEBAR
# -----------------------------------
menu = st.sidebar.selectbox(
    "Select Option",
    [
        "Register Voice",
        "Authenticate Voice",
        "Voice History",
    #    "Dataset Preview"
    ]
)

# -----------------------------------
# REGISTER VOICE
# -----------------------------------
if menu == "Register Voice":
    st.header("🎤 Register Your Voice")
    uploaded_file = st.file_uploader(
        "Upload your normal voice sample (.wav)",
        type=["wav"]
    )
    if uploaded_file:
        st.success("Voice profile registered successfully!")
        st.info("Voice Profile ID : VP1001")

# -----------------------------------
# AUTHENTICATE
# -----------------------------------
elif menu == "Authenticate Voice":
    st.header("🔐 Authenticate Voice")
    uploaded_file = st.file_uploader(
        "Upload current voice sample (.wav)",
        type=["wav"]
    )
    if uploaded_file:
        similarity = random.randint(70, 99)
        st.metric("Voice Similarity", f"{similarity}%")
        if similarity >= 90:
            st.success("✅ User Authenticated")
        elif similarity >= 80:
            st.warning("⚠ Voice changed. Possible cold detected.")
            if st.button("Create Temporary Voice Profile"):
                st.success("Temporary Voice Profile Created")
        else:
            st.error("❌ Authentication Failed")

# -----------------------------------
# HISTORY
# -----------------------------------
elif menu == "Voice History":
    st.header("📊 Voice Profile History")
    st.table({
        "Profile": ["Normal Voice", "Cold Voice", "Recovery Voice"],
        "Status": ["Active", "Temporary", "Archived"],
        "Similarity": ["100%", "84%", "95%"]
    })
